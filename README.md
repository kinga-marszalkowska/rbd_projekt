```SQL 
    RETURNS TRIGGER
    LANGUAGE 'plpgsql'
AS $$
BEGIN
    IF NEW.archived THEN
        UPDATE public.order SET status = 'canceled' 
        WHERE (status = 'paid' OR status = 'ordered') AND id IN 
        (SELECT order_game.order_id FROM public.order_game WHERE order_game.game_id = NEW.game_id);
    END IF;
    RETURN NULL;
END
$$;
```

```SQL 
    RETURNS TRIGGER
```
